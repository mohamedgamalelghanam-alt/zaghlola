    <script>
        let cart = [];

        function closeWelcome() {
            document.getElementById('welcome-overlay').style.display = 'none';
        }

        function addToCart(name, price) {
            cart.push({name, price});
            document.getElementById('cart-count').innerText = cart.length;
            alert('تم إضافة ' + name + ' للسلة');
        }

        function showCheckout() {
            if(cart.length === 0) return alert('السلة فارغة');
            let total = 0;
            let list = cart.map(item => {
                total += item.price;
                return `• ${item.name} (${item.price} EGP)`;
            }).join('<br>');
            document.getElementById('summary').innerHTML = list + `<br><strong>الإجمالي: ${total} EGP</strong>`;
            document.getElementById('checkout').style.display = 'block';
            window.location.hash = 'checkout';
        }

        function sendToWhatsApp() {
            const name = document.getElementById('name').value;
            const phone = document.getElementById('phone').value;
            const address = document.getElementById('address').value;
            if(!name || !address) return alert('برجاء إكمال البيانات');

            let msg = `*طلب جديد من موقع زغلولة*%0A`;
            msg += `*الاسم:* ${name}%0A*التليفون:* ${phone}%0A*العنوان:* ${address}%0A%0A*الطلبات:*%0A`;
            let total = 0;
            cart.forEach(i => { msg += `- ${i.name}%0A`; total += i.price; });
            msg += `%0A*الإجمالي:* ${total} EGP`;

            const myNumber = "201008836269"; 
            window.open(`https://wa.me/${myNumber}?text=${msg}`, '_blank');
        }
